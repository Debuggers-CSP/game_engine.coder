<script>
// Wire control buttons to the embedded runner via postMessage
document.addEventListener('DOMContentLoaded', () => {
    const frame = document.querySelector('.full-bleed-game iframe');
    if (!frame) return;

    const post = (action) => {
        try { frame.contentWindow.postMessage({ type: 'rpg:control', action }, '*'); } catch (e) {}
    };

    const startBtn = document.getElementById('rpg-btn-start');
    const pauseBtn = document.getElementById('rpg-btn-pause');
    const stopBtn = document.getElementById('rpg-btn-stop');
    const resetBtn = document.getElementById('rpg-btn-reset');

    if (startBtn) startBtn.addEventListener('click', () => {
        if (editor) {
            // Run the current editor code; this avoids the default turtle runner
            reloadAndRun(editor.value);
            // Ensure Pause button text/state is reset
            if (pauseBtn) {
                pauseBtn.textContent = 'Pause';
                pauseBtn.setAttribute('data-paused', 'false');
            }
        } else {
            // Fallback: no editor found, send start (should still be a no-op in embedded)
            post('start');
        }
    });

    if (pauseBtn) pauseBtn.addEventListener('click', () => {
        const paused = pauseBtn.getAttribute('data-paused') === 'true';
        if (paused) {
            post('resume');
            pauseBtn.textContent = 'Pause';
            pauseBtn.setAttribute('data-paused', 'false');
        } else {
            post('pause');
            pauseBtn.textContent = 'Resume';
            pauseBtn.setAttribute('data-paused', 'true');
        }
    });

    if (stopBtn) stopBtn.addEventListener('click', () => {
        post('stop');
        // Reset pause button state and label when stopped
        if (pauseBtn) {
            pauseBtn.textContent = 'Pause';
            pauseBtn.setAttribute('data-paused', 'false');
        }
    });

    if (resetBtn) resetBtn.addEventListener('click', () => {
        // Reset code editor to original content
        if (editor && typeof initialEditorCode !== 'undefined') {
            editor.value = initialEditorCode;
        }
        // Reset runner to a clean, stopped state
        post('reset');
        // As a safety, reload the iframe if runner doesn't respond
        setTimeout(() => {
            try { frame.src = frame.src; } catch (e) {}
        }, 300);
        // Reset Pause button state and label
        if (pauseBtn) {
            pauseBtn.textContent = 'Pause';
            pauseBtn.setAttribute('data-paused', 'false');
        }
    });

    // Live code controls
    const copyBtn = document.getElementById('code-copy');
    const editor = document.getElementById('runner-code');
    const initialEditorCode = editor ? editor.value : '';
    // Tools UI logic
    const addNpcBtn = document.getElementById('tools-add-npc');
    const npcForm = document.getElementById('npc-form');
    const npcCancel = document.getElementById('npc-cancel');
    const npcApply = document.getElementById('npc-apply');
    const npcId = document.getElementById('npc-id');
    const npcGreeting = document.getElementById('npc-greeting');
    const npcSprite = document.getElementById('npc-sprite');
    const npcScale = document.getElementById('npc-scale');
    const npcAnim = document.getElementById('npc-anim');
    const npcX = document.getElementById('npc-x');
    const npcY = document.getElementById('npc-y');
    const npcDialogues = document.getElementById('npc-dialogues');

    function showNpcForm() { npcForm.style.display = 'block'; }
    function hideNpcForm() { npcForm.style.display = 'none'; }
    if (addNpcBtn) addNpcBtn.addEventListener('click', showNpcForm);
    if (npcCancel) npcCancel.addEventListener('click', hideNpcForm);

    function ensureImport(code, identifier, path) {
        if (code.includes("import " + identifier)) return code;
        const importLine = `import ${identifier} from '${path}';`;
        const lines = code.split('\n');
        let lastImportIdx = -1;
        for (let i = 0; i < lines.length; i++) {
            if (lines[i].trim().startsWith('import ')) lastImportIdx = i;
        }
        if (lastImportIdx >= 0) {
            lines.splice(lastImportIdx + 1, 0, importLine);
            return lines.join('\n');
        }
        return importLine + '\n' + code;
    }

    function sanitizeVar(name) { return name.toLowerCase().replace(/[^a-z0-9]+/g, ''); }

    function spritePath(key) {
        switch (key) {
            case 'tux': return "/images/gamify/tux.png";
            case 'r2d2': return "/images/gamify/r2_idle.png";
            case 'chillguy': return "/images/gamify/chillguy.png";
            default: return "/images/gamify/tux.png";
        }
    }

    function buildNpcBlock(opts) {
        const varKey = sanitizeVar(opts.id || 'npc');
        const spriteVar = `sprite_src_${varKey}`;
        const greetVar = `sprite_greet_${varKey}`;
        const dataVar = `sprite_data_${varKey}`;
        const dialoguesArr = (opts.dialogues || []).map(s => s.replace(/\\/g,'\\\\').replace(/`/g,'\\`').replace(/"/g,'\\"')).map(s => `"${s}"`).join(',\n            ');
        return `// NPC data for ${opts.id}\n    const ${spriteVar} = path + "${spritePath(opts.sprite)}";\n    const ${greetVar} = "${opts.greeting}";\n    const ${dataVar} = {\n        id: '${opts.id}',\n        greeting: ${greetVar},\n        src: ${spriteVar},\n        SCALE_FACTOR: ${opts.scale},\n        ANIMATION_RATE: ${opts.anim},\n        pixels: {height: 256, width: 352},\n        INIT_POSITION: { x: ${opts.x || 'width / 2'}, y: ${opts.y || 'height / 2'}},\n        orientation: {rows: 8, columns: 11 },\n        down: {row: 5, start: 0, columns: 3 },\n        hitbox: { widthPercentage: 0.1, heightPercentage: 0.2 },\n        dialogues: [\n            ${dialoguesArr}\n        ],\n        reaction: function() {\n            if (this.dialogueSystem) {\n                this.showReactionDialogue();\n            } else {\n                console.log(${greetVar});\n            }\n        },\n        interact: function() {\n            if (this.dialogueSystem) {\n                this.showRandomDialogue();\n            }\n        }\n    };\n`;
    }

    function insertNpcIntoCode(code, opts) {
        // Ensure Npc import
        let working = ensureImport(code, 'Npc', '/assets/js/adventureGame/GameEngine/Npc.js');

        // Insert NPC block before classes definition
        const npcBlock = buildNpcBlock(opts);
        const marker = '// Define objects for this level';
        if (working.includes(marker)) {
            working = working.replace(marker, npcBlock + '\n' + marker);
        } else {
            // Fallback: insert before 'this.classes = ['
            working = working.replace(/this\.classes\s*=\s*\[/, npcBlock + '\nthis.classes = [');
        }

        // Add to classes array
        const varKey = sanitizeVar(opts.id || 'npc');
        const dataVar = `sprite_data_${varKey}`;
        if (!working.includes(`{ class: Npc, data: ${dataVar} }`)) {
            working = working.replace(/(this\.classes\s*=\s*\[)([\s\S]*?)(\])/m, (m, p1, p2, p3) => {
                const trimmed = p2.trim();
                const needsComma = trimmed && !trimmed.endsWith(',');
                const insertion = (needsComma ? ',\n' : '') + `      { class: Npc, data: ${dataVar} }`;
                return p1 + p2 + insertion + p3;
            });
        }

        // Locate the inserted block for highlighting/typing animation
        const blockStart = working.indexOf(npcBlock);
        const blockEnd = blockStart >= 0 ? (blockStart + npcBlock.length) : -1;

        return { code: working, blockText: npcBlock, blockStart, blockEnd };
    }

    function animateInsert(editorEl, finalCode, startIndex, insertedText) {
        // Prepare before/after segments from the final code snapshot
        const before = finalCode.slice(0, startIndex);
        const after = finalCode.slice(startIndex + insertedText.length);

        // Start with base (without inserted block)
        editorEl.value = before + after;
        editorEl.focus();

        // Typing animation variables
        const step = 4; // characters per tick
        let pos = 0;

        function tick() {
            // Append next chunk
            pos = Math.min(pos + step, insertedText.length);
            const typed = insertedText.slice(0, pos);
            editorEl.value = before + typed + after;

            // Highlight the typed region
            const selStart = before.length;
            const selEnd = before.length + typed.length;
            try { editorEl.setSelectionRange(selStart, selEnd); } catch (_) {}

            if (pos < insertedText.length) {
                window.requestAnimationFrame(tick);
            } else {
                // Keep highlight briefly, then clear selection
                setTimeout(() => {
                    try { editorEl.setSelectionRange(selEnd, selEnd); } catch (_) {}
                }, 600);
            }
        }

        window.requestAnimationFrame(tick);
    }

    if (npcApply) npcApply.addEventListener('click', () => {
        if (!editor) return;
        const opts = {
            id: npcId.value.trim() || 'NPC',
            greeting: npcGreeting.value.trim() || 'Hello!',

        } else {
            // Fallback: no block found, just set the code
            editor.value = result.code;
        }
        hideNpcForm();
    });

    function reloadAndRun(codeText) {
        try {
            const url = new URL(frame.src, window.location.href);
            url.searchParams.set('embed', '1');
            url.searchParams.set('autostart', '0');
            url.searchParams.set('ts', Date.now().toString());
            frame.src = url.toString();
            const onLoad = () => {
                // small delay to ensure module listeners are ready
                setTimeout(() => {
                    try { frame.contentWindow.postMessage({ type: 'rpg:run-code', code: codeText }, '*'); } catch (e) { console.error('postMessage failed', e); }
                }, 30);
                frame.removeEventListener('load', onLoad);
            };
            frame.addEventListener('load', onLoad);
        } catch (e) {
            console.error('reloadAndRun error', e);
        }
    }

    if (copyBtn && editor) copyBtn.addEventListener('click', async () => {
        try { await navigator.clipboard.writeText(editor.value); } catch (_) {}
    });
});

// Track page visit
function trackPageVisit(pageNumber) {
    let visitedPages = JSON.parse(localStorage.getItem('rpgVisitedPages') || '[]');
    if (!visitedPages.includes(pageNumber)) {
        visitedPages.push(pageNumber);
        localStorage.setItem('rpgVisitedPages', JSON.stringify(visitedPages));
    }
}

// Check if review page should be unlocked
function updateReviewLock() {
    const visitedPages = JSON.parse(localStorage.getItem('rpgVisitedPages') || '[]');
    const requiredPages = [2, 3, 4, 5];
    const allVisited = requiredPages.every(page => visitedPages.includes(page));
  
    const reviewLink = document.getElementById('review-link');
    if (reviewLink) {
        if (allVisited) {
            reviewLink.classList.remove('locked');
        } else {
            reviewLink.classList.add('locked');
        }
    }
}

// Update visited page indicators
function updateVisitedIndicators() {
    const visitedPages = JSON.parse(localStorage.getItem('rpgVisitedPages') || '[]');
    document.querySelectorAll('.nav-link').forEach(link => {
        const pageNum = parseInt(link.dataset.page);
        if (visitedPages.includes(pageNum)) {
            link.classList.add('visited');
        }
    });
}
