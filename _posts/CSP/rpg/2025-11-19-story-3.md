<script>
document.addEventListener('DOMContentLoaded', () => {
    const assets = {
        bg: {
            desert: { src: "/images/gamify/desert.png", h: 580, w: 1038 },
            alien: { src: "/images/gamify/alien_planet.png", h: 600, w: 1000 },
            clouds: { src: "/images/gamify/clouds.png", h: 720, w: 1280 }
        },
        sprites: {
            tux: { src: "/images/gamify/tux.png", h:256, w:352, rows:8, cols:11 },
            chillguy: { src: "/images/gamify/chillguy.png", h:384, w:512, rows:3, cols:4 }
        }
    };

    const ui = {
        bg: document.getElementById('bg-select'),
        pSprite: document.getElementById('player-select'),
        pX: document.getElementById('player-x'),
        nMsg: document.getElementById('npc-msg'),
        nX: document.getElementById('npc-x'),
        editor: document.getElementById('code-editor'),
        hLayer: document.getElementById('highlight-layer'),
        iframe: document.getElementById('game-iframe')
    };

    const LINE_HEIGHT = 20;

    function generateCode() {
        const bg = assets.bg[ui.bg.value];
        const p = assets.sprites[ui.pSprite.value];
        const movement = document.getElementById('movement-keys').value;
        const keypress = movement === 'arrows'
            ? '{ up: 38, left: 37, down: 40, right: 39 }'
            : '{ up: 87, left: 65, down: 83, right: 68 }';

        return `import GameControl from '/assets/js/adventureGame/GameEngine/GameControl.js';
import GameEnvBackground from '/assets/js/adventureGame/GameEngine/GameEnvBackground.js';
import Player from '/assets/js/adventureGame/GameEngine/Player.js';
import Npc from '/assets/js/adventureGame/GameEngine/Npc.js';

class CustomLevel {
  constructor(gameEnv) {
    const path = gameEnv.path;
    const height = gameEnv.innerHeight;

    const bgData = {
      name: 'custom_bg',
      src: path + "${bg.src}",
      pixels: { height: ${bg.h}, width: ${bg.w} }
    };

    const playerData = {
      id: 'Hero',
      src: path + "${p.src}",
      SCALE_FACTOR: 5,
      INIT_POSITION: { x: ${ui.pX.value}, y: height - 100 },
      pixels: { height: ${p.h}, width: ${p.w} },
      orientation: { rows: ${p.rows}, columns: ${p.cols} },
      down: { row: 0, start: 0, columns: 3 },
      keypress: ' + keypress + '
    };

    const npcData = {
      id: 'Guide',
      src: path + "/images/gamify/r2_idle.png",
      SCALE_FACTOR: 8,
      INIT_POSITION: { x: ${ui.nX.value}, y: height - 100 },
      pixels: { height: 223, width: 505 },
      orientation: { rows: 1, columns: 3 },
      down: { row: 0, start: 0, columns: 3 },
      interact: function() { alert("${ui.nMsg.value}"); }
    };

    this.classes = [
      { class: GameEnvBackground, data: bgData },
      { class: Player, data: playerData },
      { class: Npc, data: npcData }
    ];
  }
}

export { GameControl };
export const gameLevelClasses = [CustomLevel];`;
    }

    function applyShiftedHighlights(oldCode, newCode) {
        ui.hLayer.innerHTML = '';
        const oldLines = oldCode.split('\n');
        const newLines = newCode.split('\n');
        
        newLines.forEach((line, idx) => {
            // Identify if the block header has changed
            const isHeader = line.includes('name:') || line.includes('id:');
            
            if (line !== oldLines[idx] && isHeader) {
                const box = document.createElement('div');
                box.className = 'highlight-box';
                
                // IDX + 1 moves the highlight DOWN to start at 'src'
                box.style.top = ((idx + 1) * LINE_HEIGHT) + 'px';
                
                // Determine how many lines to highlight below the header
                let count = 2; // Default for background (src + pixels)
                if (line.includes("'Hero'")) count = 7; // Player properties
                if (line.includes("'Guide'")) count = 6; // NPC properties
                
                box.style.height = (count * LINE_HEIGHT) + 'px';
                ui.hLayer.appendChild(box);
            }
        });
    }

    document.getElementById('btn-confirm').addEventListener('click', () => {
        const oldCode = ui.editor.value;
        const newCode = generateCode();
        ui.editor.value = newCode;
        // Small delay to ensure browser paints text before highlight
        setTimeout(() => applyShiftedHighlights(oldCode, newCode), 10);
    });

    document.getElementById('btn-run').addEventListener('click', () => {
        ui.hLayer.innerHTML = '';
        const code = ui.editor.value;
        ui.iframe.src = ui.iframe.src; 
        ui.iframe.onload = () => {
            setTimeout(() => {
                ui.iframe.contentWindow.postMessage({ type: 'rpg:run-code', code: code }, '*');
            }, 100);
        };
    });

    ui.editor.value = generateCode();
});
</script>
