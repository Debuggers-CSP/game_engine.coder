function generatePlot() {
    const hero = plotElements.heroes[Math.floor(Math.random() * plotElements.heroes.length)];
    const villain = plotElements.villains[Math.floor(Math.random() * plotElements.villains.length)];
    const conflict = plotElements.conflicts[Math.floor(Math.random() * plotElements.conflicts.length)];
    const location = plotElements.locations[Math.floor(Math.random() * plotElements.locations.length)];
    const twist = plotElements.twists[Math.floor(Math.random() * plotElements.twists.length)];
    const item = plotElements.items[Math.floor(Math.random() * plotElements.items.length)];
    
    const resultBox = document.getElementById('result-box');
    const plotResult = document.getElementById('plot-result');
    
    plotResult.innerHTML = `
        <span class="result-label">üé≠ Your Generated Plot:</span><br><br>
        
        <strong style="color: #ffed4e;">Hero:</strong> ${hero}<br><br>
        
        <strong style="color: #ffed4e;">Villain:</strong> ${villain} who ${conflict}<br><br>
        
        <strong style="color: #ffed4e;">Key Location:</strong> ${location}<br><br>
        
        <strong style="color: #ffed4e;">Plot Twist:</strong> ${twist}<br><br>
        
        <strong style="color: #ffed4e;">Legendary Item:</strong> ${item}
    `;

}
