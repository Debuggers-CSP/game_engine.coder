    function ensureImport(code, identifier, path) {

        }
        if (lastImportIdx >= 0) {
            lines.splice(lastImportIdx + 1, 0, importLine);
            return lines.join('\n');
        }
        return importLine + '\n' + code;

    function insertNpcIntoCode(code, opts) {

        let working = ensureImport(code, 'Npc', '/assets/js/adventureGame/GameEngine/Npc.js');

        // Insert NPC block before classes definition
        const npcBlock = buildNpcBlock(opts);
        const marker = '// Define objects for this level';
        if (working.includes(marker)) {
            working = working.replace(marker, npcBlock + '\n' + marker);
        } else {
            // Fallback: insert before 'this.classes = ['
            working = working.replace(/this\.classes\s*=\s*\[/, npcBlock + '\nthis.classes = [');
        }

        // Add to classes array
        const varKey = sanitizeVar(opts.id || 'npc');
        const dataVar = `sprite_data_${varKey}`;
        if (!working.includes(`{ class: Npc, data: ${dataVar} }`)) {
            working = working.replace(/(this\.classes\s*=\s*\[)([\s\S]*?)(\])/m, (m, p1, p2, p3) => {
                const trimmed = p2.trim();
                const needsComma = trimmed && !trimmed.endsWith(',');
                const insertion = (needsComma ? ',\n' : '') + `      { class: Npc, data: ${dataVar} }`;
                return p1 + p2 + insertion + p3;
            });
        }

        // Locate the inserted block for highlighting/typing animation
        const blockStart = working.indexOf(npcBlock);
        const blockEnd = blockStart >= 0 ? (blockStart + npcBlock.length) : -1;

        return { code: working, blockText: npcBlock, blockStart, blockEnd, dataVar };
    }

    function animateInsert(editorEl, finalCode, startIndex, insertedText) {
        // Prepare before/after segments from the final code snapshot
        const before = finalCode.slice(0, startIndex);
        const after = finalCode.slice(startIndex + insertedText.length);

        // Start with base (without inserted block)
        // Clear old persistent highlight before new insertion
        clearPersistentHighlight();
        editorEl.value = before + after;
        editorEl.focus();

        // Typing animation variables
        const step = 4; // characters per tick
        let pos = 0;

        function tick() {
            // Append next chunk
            pos = Math.min(pos + step, insertedText.length);
            const typed = insertedText.slice(0, pos);
            editorEl.value = before + typed + after;

            // Highlight the typed region
            const selStart = before.length;
            const selEnd = before.length + typed.length;
            try { editorEl.setSelectionRange(selStart, selEnd); } catch (_) {}

            if (pos < insertedText.length) {
                window.requestAnimationFrame(tick);
            } else {
                // Set persistent highlight to remain visible even after blur.
                const selStart = before.length;
                const selEnd = before.length + insertedText.length;
                setPersistentHighlight(selStart, selEnd);
            }
        }

        window.requestAnimationFrame(tick);
    }

    if (npcApply) npcApply.addEventListener('click', () => {
        if (!editor) return;
        const opts = {
            id: npcId.value.trim() || 'NPC',
            greeting: npcGreeting.value.trim() || 'Hello!',
            sprite: npcSprite.value,
            scale: parseInt(npcScale.value || '8', 10),
            anim: parseInt(npcAnim.value || '50', 10),
            x: npcX.value ? npcX.value : 'width / 2',
            y: npcY.value ? npcY.value : 'height / 2',
            dialogues: npcDialogues.value.split('\n').map(s => s.trim()).filter(Boolean)
        };
        const result = insertNpcIntoCode(editor.value, opts);
        if (result.blockStart >= 0) {
            animateInsert(editor, result.code, result.blockStart, result.blockText);
            // Track change in the log and show panel
            const change = {
                type: 'npc',
                name: opts.id,
                id: sanitizeVar(opts.id || 'npc'),
                dataVar: result.dataVar,
                blockText: result.blockText,
                status: 'active'
            };
            changesLog.push(change);
            renderChangesPanel();
        } else {
            // Fallback: no block found, just set the code
            editor.value = result.code;
        }
        hideNpcForm();
    });

    function renderChangesPanel() {
        if (!changesPanel || !changesList) return;
        changesPanel.style.display = 'block';
        changesList.innerHTML = changesLog.map((c, idx) => {
            const undoDisabled = c.status !== 'active';
            const redoDisabled = c.status !== 'undone';
            return `
                <li class="change-item" data-index="${idx}">
                    <span class="change-name">NPC: ${c.name}</span>
                    <span class="change-actions">
                        <button class="change-undo" ${undoDisabled ? 'disabled' : ''} title="Remove from code">Undo</button>
                        <button class="change-redo" ${redoDisabled ? 'disabled' : ''} title="Add back to code">Add Back</button>
                    </span>
                </li>`;
        }).join('');
    }

    function removeNpcFromCode(code, change) {
        let working = code;
        // Remove the NPC block text
        if (change.blockText) {
            working = working.replace(change.blockText, '');
        }
        // Remove the classes array entry referencing this NPC (with optional leading comma)
        const classEntryRegex = new RegExp(`\n?\s*,?\s*\{\s*class:\s*Npc\s*,\s*data:\s*${change.dataVar}\s*\}`);
        working = working.replace(classEntryRegex, '');
        // Clean up dangling commas before closing bracket in classes array
        working = working.replace(/,\s*\]/g, ']');
        // Compress excessive blank lines that may result from removal
        working = working.replace(/\n{3,}/g, '\n\n');
        return working;
    }

    function reAddNpcToCode(code, change) {
        // Ensure Npc import present
        let working = ensureImport(code, 'Npc', '/assets/js/adventureGame/GameEngine/Npc.js');
        const marker = '// Define objects for this level';
        const block = change.blockText;
        if (working.includes(marker)) {
            working = working.replace(marker, block + '\n' + marker);
        } else {
            working = working.replace(/this\.classes\s*=\s*\[/, block + '\nthis.classes = [');
        }
        // Add class entry back if missing
        if (!working.includes(`{ class: Npc, data: ${change.dataVar} }`)) {
            working = working.replace(/(this\.classes\s*=\s*\[)([\s\S]*?)(\])/m, (m, p1, p2, p3) => {
                const trimmed = p2.trim();
                const needsComma = trimmed && !trimmed.endsWith(',');
                const insertion = (needsComma ? ',\n' : '') + `      { class: Npc, data: ${change.dataVar} }`;
                return p1 + p2 + insertion + p3;
            });
        }
        return working;
    }

    if (changesList) changesList.addEventListener('click', (e) => {
        const item = e.target.closest('.change-item');
        if (!item) return;
        const idx = parseInt(item.getAttribute('data-index'), 10);
        const change = changesLog[idx];
        if (!change || !editor) return;
        if (e.target.classList.contains('change-undo')) {
            if (change.status !== 'active') return;
            const updated = removeNpcFromCode(editor.value, change);
            editor.value = updated;
            // Clear highlight when NPC is removed
            clearPersistentHighlight();
            change.status = 'undone';
            renderChangesPanel();
        } else if (e.target.classList.contains('change-redo')) {
            if (change.status !== 'undone') return;
            const updated = reAddNpcToCode(editor.value, change);
            // Animate re-insert highlight
            const start = updated.indexOf(change.blockText);
            if (start >= 0) {
                animateInsert(editor, updated, start, change.blockText);
            } else {
                editor.value = updated;
            }
            change.status = 'active';
            renderChangesPanel();
        }
    });

    function reloadAndRun(codeText) {
        try {
            const url = new URL(frame.src, window.location.href);
            url.searchParams.set('embed', '1');
            url.searchParams.set('autostart', '0');
            url.searchParams.set('ts', Date.now().toString());
            frame.src = url.toString();
            const onLoad = () => {
                // small delay to ensure module listeners are ready
                setTimeout(() => {
                    try { frame.contentWindow.postMessage({ type: 'rpg:run-code', code: codeText }, '*'); } catch (e) { console.error('postMessage failed', e); }
                }, 30);
                frame.removeEventListener('load', onLoad);
            };
            frame.addEventListener('load', onLoad);
        } catch (e) {
            console.error('reloadAndRun error', e);
        }
    }

    if (copyBtn && editor) copyBtn.addEventListener('click', async () => {
        try { await navigator.clipboard.writeText(editor.value); } catch (_) {}
    });
});
</script>

// Track page visit
function trackPageVisit(pageNumber) {
    let visitedPages = JSON.parse(localStorage.getItem('rpgVisitedPages') || '[]');
    if (!visitedPages.includes(pageNumber)) {
        visitedPages.push(pageNumber);
        localStorage.setItem('rpgVisitedPages', JSON.stringify(visitedPages));
    }
}

// Check if review page should be unlocked
function updateReviewLock() {
    const visitedPages = JSON.parse(localStorage.getItem('rpgVisitedPages') || '[]');
    const requiredPages = [2, 3, 4, 5];
    const allVisited = requiredPages.every(page => visitedPages.includes(page));
  
    const reviewLink = document.getElementById('review-link');
    if (reviewLink) {
        if (allVisited) {
            reviewLink.classList.remove('locked');
        } else {
            reviewLink.classList.add('locked');
        }
    }
}

// Update visited page indicators
function updateVisitedIndicators() {
    const visitedPages = JSON.parse(localStorage.getItem('rpgVisitedPages') || '[]');
    document.querySelectorAll('.nav-link').forEach(link => {
        const pageNum = parseInt(link.dataset.page);
        if (visitedPages.includes(pageNum)) {
            link.classList.add('visited');
        }
    });
}
