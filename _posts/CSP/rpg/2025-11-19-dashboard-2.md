// Dashboard Manager
class Dashboard {
    constructor() {
        this.sections = [
            { id: 'story', name: 'Story', icon: 'üìñ', color: '#9B59B6', path: '/rpg/story' },
            { id: 'game-creator', name: 'Game', icon: 'üéÆ', color: '#3498DB', path: '/rpg/game-creator' },
            { id: 'controls', name: 'Controls', icon: 'üéØ', color: '#2ECC71', path: '/rpg/controls' },
            { id: 'systems', name: 'Systems', icon: '‚öôÔ∏è', color: '#E67E22', path: '/rpg/systems' },
            { id: 'review', name: 'Review', icon: 'üìã', color: '#E74C3C', path: '/rpg/review' },
            { id: 'tools', name: 'Tools', icon: 'üß∞', color: '#95A5A6', path: '/rpg/tools' }
        ];
        
        this.init();
    }
    
    init() {
        this.setupCardInteractions();
        this.setupTooltips();
    }
    
    setupCardInteractions() {
        const cards = document.querySelectorAll('.visual-card');
        
        cards.forEach((card, index) => {
            const section = this.sections[index];
            
            // Click effect
            card.addEventListener('click', () => {
                this.animateCardClick(card);
                
                // Navigate after animation
                setTimeout(() => {
                    window.location.href = section.path;
                }, 300);
            });
            
            // Create subtle glow effect
            card.addEventListener('mouseenter', () => {
                this.animateCardHover(card, section);
            });
        });
    }
    
    animateCardClick(card) {
        // Scale down
        card.style.transform = 'scale(0.95)';
        
        // Add glow effect
        card.style.boxShadow = `
            0 0 60px rgba(255, 215, 0, 0.6),
            0 15px 40px rgba(0, 0, 0, 0.7)
        `;
        
        // Reset after animation
        setTimeout(() => {
            card.style.transform = '';
            card.style.boxShadow = '';
        }, 300);
    }
    
    animateCardHover(card, section) {
        // Create floating particles on hover
        if (!card._particlesCreated) {
            this.createHoverParticles(card, section.color);
            card._particlesCreated = true;
        }
    }
    
    createHoverParticles(card, color) {
        const visualArea = card.querySelector('.card-visual');
        
        for (let i = 0; i < 5; i++) {
            const particle = document.createElement('div');
            particle.style.cssText = `
                position: absolute;
                width: 4px;
                height: 4px;
                background: ${color};
                border-radius: 50%;
                pointer-events: none;
                opacity: 0;
                filter: blur(1px);
            `;
            
            visualArea.appendChild(particle);
            
            // Animate particle
            const x = Math.random() * 100;
            const y = Math.random() * 100;
            const duration = Math.random() * 1 + 1;
            
            particle.style.left = `${x}%`;
            particle.style.top = `${y}%`;
            particle.style.animation = `
                particleFloat ${duration}s ease-in-out infinite
            `;
        }
        
        // Add animation keyframes
        if (!document.getElementById('particle-animations')) {
            const style = document.createElement('style');
            style.id = 'particle-animations';
            style.textContent = `
                @keyframes particleFloat {
                    0%, 100% { 
                        opacity: 0; 
                        transform: translateY(0) rotate(0deg); 
                    }
                    50% { 
                        opacity: 0.6; 
                        transform: translateY(-20px) rotate(180deg); 
                    }
                }
            `;
            document.head.appendChild(style);
        }
    }
    
    setupTooltips() {
        // Tooltips are already handled by CSS
        // Add any additional tooltip logic here if needed
    }
    
    // Public methods
    saveAll() {
        // Create visual feedback
        const saveBtn = document.querySelector('.action-item:nth-child(1) .action-icon');
        saveBtn.textContent = '‚úÖ';
        saveBtn.style.background = 'rgba(46, 204, 113, 0.2)';
        saveBtn.style.borderColor = '#2ECC71';
        
        setTimeout(() => {
            saveBtn.textContent = 'üíæ';
            saveBtn.style.background = '';
            saveBtn.style.borderColor = '';
        }, 1000);
    }
    
    exportProject() {
        // Visual feedback
        const exportBtn = document.querySelector('.action-item:nth-child(2) .action-icon');
        const originalText = exportBtn.textContent;
        exportBtn.textContent = '‚è≥';
        exportBtn.style.background = 'rgba(52, 152, 219, 0.2)';
        exportBtn.style.borderColor = '#3498DB';
        
        setTimeout(() => {
            exportBtn.textContent = 'üì§';
            exportBtn.style.background = '';
            exportBtn.style.borderColor = '';
            alert('Export complete!');
        }, 1500);
    }
    
    showHelp() {
        // Simple help modal
        const helpContent = `
            RPG Creator Dashboard
            ---------------------
            ‚Ä¢ Click any card to open
            ‚Ä¢ Hover for tooltips
            ‚Ä¢ Progress shown in bottom right
            ‚Ä¢ All sections are accessible
            ‚Ä¢ Save frequently
            ‚Ä¢ Export when ready
        `;
        
        alert(helpContent);
    }
    
    refreshDashboard() {
        // Visual feedback
        const refreshBtn = document.querySelector('.action-item:nth-child(4) .action-icon');
        refreshBtn.style.animation = 'spin 1s linear';
        
        setTimeout(() => {
            refreshBtn.style.animation = '';
            location.reload();
        }, 1000);
        
        // Add spin animation
        if (!document.getElementById('spin-animation')) {
            const style = document.createElement('style');
            style.id = 'spin-animation';
            style.textContent = `
                @keyframes spin {
                    0% { transform: rotate(0deg); }
                    100% { transform: rotate(360deg); }
                }
            `;
            document.head.appendChild(style);
        }
    }
}
